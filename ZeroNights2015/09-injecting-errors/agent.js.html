1 'use strict';
 2 
 3 const AF_INET = 2;
 4 const AF_INET6 = 30;
 5 const ECONNREFUSED = 61;
 6 
 7 ['connect', 'connect$NOCANCEL'].forEach(funcName => {
 8   const connect = new NativeFunction(
 9     Module.findExportByName('libsystem_kernel.dylib', funcName),
10     'int',
11     ['int', 'pointer', 'int']);
12   Interceptor.replace(connect, new NativeCallback((socket, address, addressLen) => {
13     const family = Memory.readU8(address.add(1));
14     if (family == AF_INET || family == AF_INET6) {
15       const port = (Memory.readU8(address.add(2)) << 8) | Memory.readU8(address.add(3));
16 
17       let ip = '';
18       if (family == AF_INET) {
19         for (let offset = 4; offset != 8; offset++) {
20           if (ip.length > 0)
21             ip += '.';
22           ip += Memory.readU8(address.add(offset));
23         }
24       } else {
25         for (let offset = 8; offset !== 24; offset += 2) {
26           if (ip.length > 0)
27             ip += ':';
28           ip += toHex(Memory.readU8(address.add(offset))) +
29               toHex(Memory.readU8(address.add(offset + 1)));
30         }
31       }
32 
33       console.log('connect() family=' + family + ' ip=' + ip + ' port=' + port);
34       if (port === 80 || port === 443 || port === 4070) {
35         console.log('  blocking!');
36         this.errno = ECONNREFUSED;
37         return -1;
38       } else {
39         console.log('  accepting!');
40         return connect(socket, address, addressLen);
41       }
42     } else {
43       return connect(socket, address, addressLen);
44     }
45   }, 'int', ['int', 'pointer', 'int']));
46 
47   send('ready');
48 });
49 
50 function toHex(v) {
51   let result = v.toString(16);
52   if (result.length === 1)
53     result = '0' + result;
54   return result;
55 }
